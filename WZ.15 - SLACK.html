Enriquecer alertas con Gemini

Requisito previo: https://youtu.be/7gzxZ3VPJrM?si=0c1UCJJS6ha_7-OP

A√±adir integraci√≥n al archivo de configuraci√≥n.

  <name>custom-gemini-slack</name>
  <rule_id>100210</rule_id>
  <hook_url>WEBHOOK</hook_url>

Crear custom-gemini-slack.

sudo nano /var/ossec/integrations/custom-gemini-slack

Copiar y pegar.

#!/bin/bash
# Enviar a Slack la alerta enriquecida por Gemini (rule 100210)

alert_file="$1"
slack_webhook="$3"

# Fallback por si alg√∫n d√≠a cambia la implementaci√≥n
if [ -z "$slack_webhook" ] && [ -n "$OSSEC_INTEGRATION_URL" ]; then
  slack_webhook="$OSSEC_INTEGRATION_URL"
fi

log="/var/ossec/logs/custom-gemini-slack.log"

echo "[custom-gemini-slack] alert_file=$alert_file webhook=${slack_webhook:0:60}..." >> "$log"

if [ -z "$slack_webhook" ]; then
  echo "[custom-gemini-slack] ERROR: sin webhook" >> "$log"
  exit 1
fi

if [ ! -f "$alert_file" ]; then
  echo "[custom-gemini-slack] ERROR: alerta no existe" >> "$log"
  exit 1
fi

json="$(cat "$alert_file")"

# ---------------- Campos b√°sicos -----------------

rule_id=$(echo "$json"        | jq -r '.rule.id // "N/A"')
rule_desc=$(echo "$json"      | jq -r '.rule.description // "N/A"')
rule_level_gemini=$(echo "$json" | jq -r '.rule.level // "N/A"')

# Nivel de la alerta ORIGINAL (la que analiz√≥ Gemini, ej. 87105 nivel 12)
orig_level=$(echo "$json" | jq -r '.data.gemini.source.level // .rule.level // "N/A"')

timestamp=$(echo "$json"      | jq -r '.timestamp // "N/A"')
agent_name=$(echo "$json"     | jq -r '.agent.name // "N/A"')
agent_ip=$(echo "$json"       | jq -r '.agent.ip // "-"')
manager_name=$(echo "$json"   | jq -r '.manager.name // "N/A"')

gemini_summary=$(echo "$json" | jq -r '.data.gemini.summary // ""')
gemini_short=$(echo "$json"   | jq -r '.data.gemini.summary_short // ""')

location=$(echo "$json"       | jq -r '.data.gemini.source.location // "N/A"')

# Ruta de archivo: primero source.path, si no, intentar extraer de la descripci√≥n
path=$(echo "$json"           | jq -r '.data.gemini.source.path // ""')
if [ -z "$path" ] || [ "$path" = "null" ]; then
  desc_src=$(echo "$json" | jq -r '.data.gemini.source.description // .rule.description // ""')
  # ej: "VirusTotal: Alert - /root/suspicious-file.exe - 66 engines detected this file"
  path=$(echo "$desc_src" \
    | sed -n 's/.*Alert - \(.*\) - [0-9][0-9]* engines.*/\1/p')
fi
[ -z "$path" ] && path="N/D"

vt_link=$(echo "$json"        | jq -r '.data.virustotal.permalink // "-"')

# MITRE: IDs con enlaces a ATT&CK, en una sola l√≠nea
mitre_links=$(echo "$json" | jq -r '
  .data.gemini.source.mitre.id // empty
  | map("<https://attack.mitre.org/techniques/" + . + "/|" + . + ">")
  | join(", ")
')
[ -z "$mitre_links" ] && mitre_links="N/A"

# Limitar longitud del resumen para no romper Slack
[ ${#gemini_summary} -gt 3000 ] && gemini_summary="${gemini_summary:0:3000}..."

# Arreglar markdown de Gemini: **texto** -> *texto* (Slack)
formatted_summary=$(printf '%s\n' "$gemini_summary" | sed 's/\*\*/\*/g')

# ------------ Crear mensaje de Slack (mrkdwn con emojis) -----------------

body=$(cat <<EOF
:rotating_light: $rule_desc

*‚öôÔ∏è Nivel original:* $orig_level    *(regla origen analizada por Gemini)*  
*üß© Regla Gemini:* $rule_id (nivel $rule_level_gemini)  

*üñ•Ô∏è Agente:* $agent_name (<$agent_ip|$agent_ip>)  
*üìç Location:* \`$location\`  
*üéØ MITRE:* $mitre_links  
*üìÅ Ruta archivo:* \`$path\`  
*üß™ VirusTotal:* <$vt_link|Ver an√°lisis en VirusTotal>

*üß† Resumen generado por Gemini:*  
$formatted_summary

üë®‚Äçüíª *Manager:* $manager_name  
üïí *Timestamp:* $timestamp
EOF
)

payload=$(printf '%s' "$body" | jq -Rs '{text: .}')

echo "[custom-gemini-slack] Enviando alerta enriquecida..." >> "$log"

tmp="/tmp/custom-gemini-slack.out.$$"

code=$(curl -s \
  -o "$tmp" \
  -w "%{http_code}" \
  -X POST \
  -H "Content-type: application/json" \
  --data "$payload" \
  "$slack_webhook")

echo "[custom-gemini-slack] Slack HTTP code: $code" >> "$log"
echo "[custom-gemini-slack] Slack body: $(cat $tmp)" >> "$log"

rm -f "$tmp"

exit 0
